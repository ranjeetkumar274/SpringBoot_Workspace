let done = JSON.parse(localStorage.getItem('dsa_done') || '{}');
let currentFilter = 'all';
let searchQuery = '';

// ─────────────────────────────────────────────
// HELPERS
// ─────────────────────────────────────────────
function problemId(sectionId, groupLabel, name) {
  return `${sectionId}::${groupLabel}::${name}`;
}

function diffFromGroup(diff) {
  if (diff === 'easy') return 'easy';
  if (diff === 'medium') return 'medium';
  return 'hard';
}

function allProblems() {
  const probs = [];
  for (const phase of ROADMAP) {
    for (const sec of phase.sections) {
      for (const grp of sec.groups) {
        for (const p of grp.problems) {
          probs.push({ id: problemId(sec.id, grp.label, p), diff: grp.diff, name: p, sectionId: sec.id });
        }
      }
    }
  }
  return probs;
}

function updateGlobalProgress() {
  const all = allProblems();
  const total = all.length;
  const doneCount = all.filter(p => done[p.id]).length;
  const pct = total ? Math.round((doneCount / total) * 100) : 0;
  document.getElementById('bar-fill').style.width = pct + '%';
  document.getElementById('pct-label').textContent = pct + '%';
  document.getElementById('h-done').textContent = doneCount;
  document.getElementById('h-total').textContent = total;
  document.getElementById('h-pct').textContent = pct + '%';
}

function saveDone() {
  localStorage.setItem('dsa_done', JSON.stringify(done));
}

// ─────────────────────────────────────────────
// RENDER
// ─────────────────────────────────────────────
function render() {
  const container = document.getElementById('roadmap');
  container.innerHTML = '';
  let anyVisible = false;
  let lastPhase = -1;

  for (const phase of ROADMAP) {
    if (phase.phase !== lastPhase) {
      lastPhase = phase.phase;
      const div = document.createElement('div');
      div.className = `phase-divider phase-${phase.phase}`;
      div.innerHTML = `<div class="phase-label">${phase.phaseName}</div><div class="phase-line"></div>`;
      container.appendChild(div);
    }

    for (const sec of phase.sections) {
      const card = renderSection(sec);
      if (card) { container.appendChild(card); anyVisible = true; }
    }
  }

  const noRes = document.getElementById('no-results');
  noRes.classList.toggle('visible', !anyVisible);
  updateGlobalProgress();
  updateRings();
}

function renderSection(sec) {
  // Collect all problems for this section
  const allProbs = [];
  for (const grp of sec.groups) {
    for (const p of grp.problems) {
      allProbs.push({ id: problemId(sec.id, grp.label, p), diff: grp.diff, name: p, group: grp.label });
    }
  }

  // Filter
  let visibleProbs = allProbs;
  if (searchQuery) {
    visibleProbs = allProbs.filter(p => p.name.toLowerCase().includes(searchQuery));
  }
  if (currentFilter === 'done') visibleProbs = visibleProbs.filter(p => done[p.id]);
  if (currentFilter === 'unsolved') visibleProbs = visibleProbs.filter(p => !done[p.id]);
  if (currentFilter === 'easy') visibleProbs = visibleProbs.filter(p => p.diff === 'easy');
  if (currentFilter === 'medium') visibleProbs = visibleProbs.filter(p => p.diff === 'medium');
  if (currentFilter === 'hard') visibleProbs = visibleProbs.filter(p => p.diff === 'hard');

  if (searchQuery && visibleProbs.length === 0) return null;

  const doneCount = allProbs.filter(p => done[p.id]).length;
  const easyCount = allProbs.filter(p => p.diff === 'easy').length;
  const medCount = allProbs.filter(p => p.diff === 'medium').length;
  const hardCount = allProbs.filter(p => p.diff === 'hard').length;

  const card = document.createElement('div');
  card.className = 'section-card';
  card.dataset.id = sec.id;

  // Build groups for body (filtered)
  const filteredGroups = sec.groups.map(grp => {
    const probs = (searchQuery || ['done','unsolved','easy','medium','hard'].includes(currentFilter))
      ? grp.problems.filter(p => visibleProbs.some(vp => vp.name === p && vp.group === grp.label))
      : grp.problems;
    return { ...grp, filteredProblems: probs };
  }).filter(g => g.filteredProblems.length > 0);

  const circumference = 2 * Math.PI * 13;
  const offset = circumference - (doneCount / allProbs.length) * circumference;

  card.innerHTML = `
    <div class="section-header" role="button" tabindex="0" aria-expanded="false">
      <div class="section-icon" style="background: ${sec.color}18; color: ${sec.color};">${sec.icon}</div>
      <div class="section-title-wrap">
        <div class="section-title">${sec.title}</div>
        <div class="section-meta">${sec.meta}</div>
      </div>
      <div class="section-badges">
        ${easyCount ? `<div class="badge badge-easy">${easyCount}E</div>` : ''}
        ${medCount ? `<div class="badge badge-medium">${medCount}M</div>` : ''}
        ${hardCount ? `<div class="badge badge-hard">${hardCount}H</div>` : ''}
        <div class="badge badge-count">${allProbs.length} total</div>
      </div>
      <div class="section-progress-ring" data-section="${sec.id}" data-total="${allProbs.length}" data-done="${doneCount}">
        <svg class="ring-svg" width="34" height="34" viewBox="0 0 34 34">
          <circle class="ring-bg" cx="17" cy="17" r="13" />
          <circle class="ring-fill" cx="17" cy="17" r="13"
            stroke="${sec.color}"
            stroke-dasharray="${circumference}"
            stroke-dashoffset="${offset}" />
        </svg>
        <div class="ring-text">${doneCount}/${allProbs.length}</div>
      </div>
      <div class="chevron">▼</div>
    </div>
    <div class="section-body">
      ${sec.theory ? `<div class="theory-block"><span class="theory-icon">📖</span><div class="theory-text"><strong>Theory:</strong> ${sec.theory}</div></div>` : ''}
      ${renderGroups(filteredGroups, sec)}
      ${sec.tip ? `<div class="tip-block"><span class="tip-icon">💡</span><div class="tip-text">${sec.tip}</div></div>` : ''}
    </div>
  `;

  const header = card.querySelector('.section-header');
  const body = card.querySelector('.section-body');

  header.addEventListener('click', () => toggleSection(card, header, body));
  header.addEventListener('keydown', e => { if (e.key === 'Enter' || e.key === ' ') { e.preventDefault(); toggleSection(card, header, body); }});

  // Auto-open if search or filter is active
  if (searchQuery || currentFilter !== 'all') {
    card.classList.add('open');
    body.classList.add('visible');
    header.setAttribute('aria-expanded', 'true');
  }

  return card;
}

function renderGroups(groups, sec) {
  return groups.map(grp => `
    <div class="subtopic-group">
      <div class="subtopic-group-label">${grp.label}</div>
      <div class="problems-grid">
        ${grp.filteredProblems.map(p => {
          const id = problemId(sec.id, grp.label, p);
          const isDone = !!done[id];
          return `
            <div class="problem-chip ${isDone ? 'done' : ''}" data-prob-id="${id}" data-diff="${grp.diff}">
              <div class="check-circle">${isDone ? '✓' : ''}</div>
              <span class="prob-name">${p}</span>
              <div class="diff-dot diff-${grp.diff}"></div>
            </div>
          `;
        }).join('')}
      </div>
    </div>
  `).join('');
}

function toggleSection(card, header, body) {
  const isOpen = card.classList.contains('open');
  card.classList.toggle('open');
  body.classList.toggle('visible');
  header.setAttribute('aria-expanded', !isOpen);
}

function updateRings() {
  document.querySelectorAll('.section-progress-ring').forEach(ring => {
    const total = parseInt(ring.dataset.total);
    const secId = ring.dataset.section;
    // Count done for this section
    let doneCount = 0;
    for (const phase of ROADMAP) {
      for (const sec of phase.sections) {
        if (sec.id === secId) {
          for (const grp of sec.groups) {
            for (const p of grp.problems) {
              if (done[problemId(sec.id, grp.label, p)]) doneCount++;
            }
          }
        }
      }
    }
    ring.dataset.done = doneCount;
    const circumference = 2 * Math.PI * 13;
    const offset = circumference - (doneCount / total) * circumference;
    const fill = ring.querySelector('.ring-fill');
    if (fill) fill.style.strokeDashoffset = offset;
    const text = ring.querySelector('.ring-text');
    if (text) text.textContent = `${doneCount}/${total}`;
  });
}

document.getElementById('roadmap').addEventListener('click', e => {
  const chip = e.target.closest('.problem-chip');
  if (!chip) return;
  const id = chip.dataset.probId;
  done[id] = !done[id];
  saveDone();
  // Toggle classes in-place for snappy feel
  chip.classList.toggle('done', done[id]);
  const circle = chip.querySelector('.check-circle');
  circle.textContent = done[id] ? '✓' : '';
  const name = chip.querySelector('.prob-name');
  updateGlobalProgress();
  updateRings();
});

document.querySelectorAll('.filter-btn').forEach(btn => {
  btn.addEventListener('click', () => {
    currentFilter = btn.dataset.filter;
    document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    render();
  });
});

document.getElementById('search-input').addEventListener('input', e => {
  searchQuery = e.target.value.trim().toLowerCase();
  render();
});
